{{/* Span Short Code By: Steve Kostecke k0stk@arrl.net Named Arguments: "body" - span body; overridden by .Inner if closing tag is used "class" - will be added to the span div class definition Most useful value here is "alert" for a red span "expires" - span expiration date, only YYYY-MM-DD is used "publish" - span publication date, only YYYY-MM-DD is used */}} {{ $body := "" }} {{ $class := "" }} {{ with .Get "body" }}{{ $body = . }}{{ end }} {{ with .Inner }}{{ $body = . }}{{ end }} {{ with .Get "expires" }} {{ $today := now.Format "2006-01-02" }} {{ $expires := dateFormat "2006-01-02" ( . ) }} {{ if gt $today $expires }} {{ $body = " " }} {{ end }} {{ end }} {{ with .Get "publish" }} {{ $today := now.Format "2006-01-02" }} {{ $publish := dateFormat "2006-01-02" ( . ) }} {{ if lt $today $publish }} {{ $body = " " }} {{ end }} {{ end }} {{ with .Get "class" }} {{ $class = . }} {{ end }} {{ if gt " " $body }} {{ $body }} {{ end }}
